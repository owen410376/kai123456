<script>
  function startLoading() {
    var loadingOverlay = document.querySelector('.loading-overlay');
    var playButton = document.querySelector('.play-button');
    var backgroundOverlay = document.querySelector('.background-overlay');

    playButton.classList.add('clicked'); // 添加点击动画类

    setTimeout(function() {
      playButton.style.display = 'none';
      loadingOverlay.style.display = 'flex';
      backgroundOverlay.style.backgroundColor = 'black'; // 设置背景颜色为黑色

      setTimeout(function() {
        document.querySelector('.loading-overlay').style.display = 'none';
        document.body.style.overflow = 'auto';
        jumpToGame();
      }, 3000);
    }, 300); // 等待点击动画完成后开始加载
  }
  function jumpToGame() {
    var gameUrl = "http://localhost:63342/Web/math.html?_ijt=c0v6s0tarigh9b1r5r0fibd6bm&_ij_reload=RELOAD_ON_SAVE";
    window.location.href = gameUrl;
  }
